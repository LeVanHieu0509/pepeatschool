import { getPagesUnderRoute } from 'nextra/context';
import Link from 'next/link';
import { useRouter } from 'next/router';
import filterRouteLocale from "nextra/filter-route-locale";

export function BlogHeader() {
  return (
    <div className="max-w-screen-lg mx-auto pt-4 pb-8 border-b border-gray-400 border-opacity-20 text-center">
      <h1>
        <span className="font-bold leading-tight lg:text-5xl">Cộng đồng của Pepe at school</span>
      </h1>
      <p className="text-center text-gray-500 dark:text-gray-400 font-space-grotesk">
        Bài viết, tin tức mới nhất dành cho cộng đồng
      </p>
    </div>
  );
}

export function BlogIndex() {
  // debug the function here
  const { locale, defaultLocale } = useRouter();
  const blogArticles = filterRouteLocale(
    getPagesUnderRoute("/blog"),
    locale,
    defaultLocale,
  );

const sortedPages = blogArticles.sort((a, b) => {
return (
new Date(b.frontMatter?.date || 0).getTime() -
new Date(a.frontMatter?.date || 0).getTime()
);
});

const mapPages = sortedPages.map((page) => {
return (

<li key={page.route}  class="relative bg-white flex flex-col justify-between border rounded shadow-md hover:shadow-teal-400">

<img
  class="rounded relative w-full object-cover aspect-video"
  src={page.meta?.imgUrl || page.frontMatter?.imgUrl || page.name}
  alt="Writey A.I"
  loading="lazy"
/>

          <div class="flex flex-col justify-beetween gap-1 px-2 py-2">
            <a
              href={page.route}
              class="flex justify-center items-center text-xl font-semibold text-teal-700 hover:text-teal-800 two-lines text-ellipsis">
              {page.meta?.title || page.frontMatter?.title || page.name}
            </a>

            {page.frontMatter?.date ? (
              <p className="opacity-50 text-xs mt-2">{page.frontMatter.date}</p>
            ) : null}

            <p class="text-gray-600 two-lines">
              {page.frontMatter?.description}{" "}
            </p>

            <ul class="flex flex-wrap items-center justify-start text-xs gap-2">
              <li
                title="Pricing type"
                class="flex items-center cursor-pointer gap-0.5 bg-gray-100 text-black px-2 py-0.5 rounded-full">
                <Link href={page.route} tabIndex="-1">
                  {"Đọc ngay →"}
                </Link>
              </li>
            </ul>
          </div>
        </li>

); });

return <ul class="grid gap-8 sm:grid-cols-2 lg:grid-cols-3 p-2 pt-4 xl:p-5">
{mapPages}</ul>;
}

<BlogHeader />
<BlogIndex />
